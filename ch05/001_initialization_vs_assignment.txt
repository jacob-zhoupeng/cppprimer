// 初始化过程为对象提供了初值，【只能一次】
int ival = 1024;
int *pi = 0;

// 赋值则是用一个新值覆盖对象的当前值，【可以多次】
ival = 2048;
pi = &ival;
--------------------------------------------------------------

// *赋值和初始化比较*
// 赋值
ival = 3.1415926; // OK，编译器会试着隐式的将右操作数转换
                  // 成被赋值对象的类型，赋值成功，并给出警告
pi = ival; // Error, 没有默认隐式转换规则，编译器报错。

// 初始化
int ival = 3.1415926; // OK，同上
int* pi = ival; // Error，同上
